a = 0x01;
c = 0x01;
b = 0x00;
h = 0x00;
f = 0x00;
d = 0x00;

for ( int i = 0 ; i < 7; i++ )
{
	for ( int j = 0 ; j < 7; j++ )
	{
		d = a|b;
		PORTC = d;
		a = a << 1;
	}
	
	for ( int j = 0 ; j < 7; j++ )
	{
		f = c|h;
		PORTC = f;
		c = c >> 1;
	}
	
	b = b << 1|0x01;
	h = h >> 1|0x80;
}
